  <script>
    // ======= EDIT THIS ARRAY: add your downloadable files here ========
    // url: direct link to file (relative or absolute)
    // name: visible name
    // size: optional human-readable size
    // tags: optional array to help search/filter
    const files = [
      {url: './files/example1.zip', name: 'Example Pack - v1.0', size: '24 MB', tags:['pack','demo']},
      {url: './files/manual.pdf', name: 'User Manual', size: '1.2 MB', tags:['pdf','manual']},
      {url: 'https://example-cdn.com/downloads/game-demo.exe', name: 'Game Demo (installer)', size: '120 MB', tags:['game','demo']}
    ];
    // ==================================================================

    const grid = document.getElementById('grid');
    const q = document.getElementById('q');
    const count = document.getElementById('count');
    const clearBtn = document.getElementById('clear');

    function renderList(filter=''){
      const f = filter.trim().toLowerCase();
      const visible = files.filter(file => {
        if(!f) return true;
        const hay = (file.name + ' ' + (file.tags||[]).join(' ')).toLowerCase();
        return hay.includes(f);
      });
      grid.innerHTML = '';
      count.textContent = visible.length + ' file' + (visible.length===1? '':'s') + ' available';

      visible.forEach(file => {
        const card = document.createElement('article');
        card.className = 'card';
        card.innerHTML = `
          <div style="display:flex;justify-content:space-between;align-items:center">
            <div style="font-weight:700">${escapeHtml(file.name)}</div>
            <div class="tag">${file.size || 'â€”'}</div>
          </div>
          <div class="meta">${escapeHtml((file.tags||[]).join(', '))}</div>
          <div class="btn-row">
            <a class="btn btn-download" href="${encodeURI(file.url)}" download target="_blank" rel="noopener noreferrer">Download</a>
            <a class="btn btn-link" href="${encodeURI(file.url)}" target="_blank" rel="noopener noreferrer">Open</a>
            <div style="flex:1"></div>
          </div>
        `;
        grid.appendChild(card);
      });
    }

    function escapeHtml(str){
      return String(str||'').replace(/[&<>\"]/g, s=>({ '&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;' }[s]));
    }

    q.addEventListener('input', ()=>renderList(q.value));
    clearBtn.addEventListener('click', ()=>{ q.value=''; renderList(''); q.focus(); });

    // initial render
    renderList('');

    // Optional: add keyboard shortcut '/' to focus search
    window.addEventListener('keydown', e => {
      if(e.key === '/' && document.activeElement !== q){ e.preventDefault(); q.focus(); }
    });
  </script>
